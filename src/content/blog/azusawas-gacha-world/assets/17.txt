using UnityEngine;
using UnityEngine.UI;

public class BackgroundScroller : MonoBehaviour
{
    public RawImage image;
    public float x,
        y;

    void Update()
    {
        image.uvRect = new Rect(
            image.uvRect.position + new Vector2(x, y) * Time.deltaTime,
            image.uvRect.size
        );
    }
}
